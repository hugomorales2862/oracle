QUERY BIBLIOTECA 1
select lib_titulo, edi_nombre, aut_nombre1, gen_descripcion
from libros, editorial, genero, autor
where lib_editorial=edi_codigo and lib_genero=gen_codigo and lib_autor=aut_codigo
order by lib_titulo, lib_autor, lib_editorial, lib_genero

QUERY BIBLIOTECA 7.
select per_nombre1, per_apellido1, per_edad, per_sexo
from bib_personal
where per_edad < 18
group by per_sexo, per_edad, per_nombre1, per_apellido1 
order by per_sexo

QUERY BIBLIOTECA 8.
select  per_cod, per_nombre1, per_apellido1, pre_codigo
from bib_prestamo, bib_personal
where pre_codigo = per_cod
and pre_fec_fin >= "2015-12-02"

QUERY BIBLIOTECA 9.
select lib_titulo, lib_estado from bib_libros
group by lib_estado, lib_titulo
order by lib_estado

QUERY PRUBLICIDAD 6.

QUERY PRUBLICIDAD 7.

FROM pub_cliente, pub_factura, pub_detalle_factura, pub_anuncio,
